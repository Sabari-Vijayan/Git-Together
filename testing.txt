DROP TRIGGER IF EXISTS trg_follow_ledger_insert ON follow_ledger;

-- =========================================================================
-- SECTION 2: FIX THE 500 SELECT ERROR (RLS RECURSION FIX)
-- =========================================================================

CREATE OR REPLACE FUNCTION is_member_of_room(r_id UUID, u_id UUID)
RETURNS BOOLEAN AS $$
BEGIN
  RETURN EXISTS (
    SELECT 1 FROM room_participants
    WHERE room_id = r_id AND user_id = u_id
  );
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;

DROP POLICY IF EXISTS "Allow room participants to view others" ON room_participants;

CREATE POLICY "Allow room participants to view others" ON room_participants
  FOR SELECT TO authenticated USING (
    is_member_of_room(room_id, auth.uid())
  );


-- =========================================================================
-- SECTION 3: ENABLE SUPABASE REALTIME (ONLY FOR ROOMS & PARTICIPANTS)
-- =========================================================================

-- Remove follow_ledger from publication if it exists
DO $$
BEGIN
  IF EXISTS (
    SELECT 1 FROM pg_publication_tables 
    WHERE pubname = 'supabase_realtime' AND tablename = 'follow_ledger'
  ) THEN
    ALTER PUBLICATION supabase_realtime DROP TABLE follow_ledger;
  END IF;
END $$;

DO $$
BEGIN
  IF NOT EXISTS (
    SELECT 1 FROM pg_publication_tables 
    WHERE pubname = 'supabase_realtime' AND tablename = 'rooms'
  ) THEN
    ALTER PUBLICATION supabase_realtime ADD TABLE rooms;
  END IF;

  IF NOT EXISTS (
    SELECT 1 FROM pg_publication_tables 
    WHERE pubname = 'supabase_realtime' AND tablename = 'room_participants'
  ) THEN
    ALTER PUBLICATION supabase_realtime ADD TABLE room_participants;
  END IF;
END $$;

-- SECTION 5: INACTIVITY CLEANUP FUNCTIONS
-- =========================================================================

CREATE OR REPLACE FUNCTION cleanup_inactive_rooms()
RETURNS VOID AS $$
BEGIN
  UPDATE rooms
  SET status = 'concluded'::room_status
  WHERE status IN ('waiting'::room_status, 'active'::room_status)
    AND created_at < (now() - INTERVAL '1 minute')
    AND NOT EXISTS (
      SELECT 1 FROM room_participants rp
      WHERE rp.room_id = rooms.id
    );
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;

-- Run this cleanup statement manually or trigger via cron
SELECT cleanup_inactive_rooms();
